280  {

282 
283  auto dataSize = sizeof(T);
284  static auto msgSize = sizeof(Message);
285  XY_ASSERT(dataSize < 128, "message size exceeds 128 bytes"); //limit custom data to 128 bytes
286  XY_ASSERT(m_pendingBuffer.size() - (m_inPointer - m_pendingBuffer.data()) > (dataSize + msgSize), "buffer overflow " + std::to_string(m_pendingCount)); //make sure we have enough room in the buffer
287  XY_WARNING(m_pendingBuffer.size() - (m_inPointer - m_pendingBuffer.data()) < 128, "Messagebus buffer is heavily contended!");
288 
289  Message* msg = new (m_inPointer)Message();
290  m_inPointer += msgSize;
291  msg->id = id;
292  msg->m_dataSize = dataSize;
293  msg->m_data = new (m_inPointer)T();
294  m_inPointer += dataSize;
295 
296  m_pendingCount++;
297  return static_cast<T*>(msg->m_data);
298  }
