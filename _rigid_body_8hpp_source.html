151  template <typename T>
152  T* addCollisionShape(const T& cs)
153  {
154  static_assert(std::is_base_of<CollisionShape, T>::value, "Can only add shapes of collision type");
155  m_collisionShapes.emplace_back(std::make_unique<T>(cs));
156  auto& newShape = m_collisionShapes.back();
157  if (m_body)
158  {
159  newShape->m_fixture = m_body->CreateFixture(&newShape->m_fixtureDef);
160  newShape->m_fixture->SetUserData(newShape.get());
161  newShape->destructionCallback = std::bind(&RigidBody::removeCollisionShape, this, _1, true);
162  newShape->registerCallbacks();
163  }
164  else
165  {
166  m_pendingShapes.push_back(newShape.get());
167  }
168  return dynamic_cast<T*>(newShape.get());
169  }

179  template <typename T>
180  T* addJoint(const T& joint)
181  {
182  static_assert(std::is_base_of<Joint, T>::value, "Must be of Joint type");
183 
184  XY_ASSERT(joint.getRigidBodyA() != this, "Cannot attach body to self");
185 
186  m_joints.emplace_back(std::make_unique<T>(joint));
187  auto& newJoint = m_joints.back();
188  newJoint->setRigidBodyB(this);
189  if (m_body)
190  {
191  newJoint->m_joint = m_body->GetWorld()->CreateJoint(newJoint->getDefinition());
192  newJoint->m_joint->SetUserData(newJoint.get());
193  newJoint->destructionCallback = std::bind(&RigidBody::removeJoint, this, _1, true);
194  }
195  else
196  {
197  m_pendingJoints.push_back(newJoint.get());
198  }
199  return dynamic_cast<T*>(newJoint.get());
200  }
