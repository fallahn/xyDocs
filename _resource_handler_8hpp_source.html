75  template<class T>
76  ResourceHandle load(const std::string& path)
77  {
78  if (m_resourcePaths.count(path) != 0)
79  {
80  return m_resourcePaths[path];
81  }
82 
83  //Find the correct loader for the type
84  std::type_index ti = typeid(T);
85 
86  XY_ASSERT(m_loaders.count(ti), std::string("Resource loader not found for ") + std::string(ti.name()));
87 
88  m_resources.emplace_back(m_loaders[ti].loader(path));
89 
90  //If it wasn't loaded, use fallback
91  if (!m_resources.back().has_value())
92  {
93  //check to see if we already cached the fallback
94  if (m_fallbackHandles.count(ti) == 0)
95  {
96  m_resources.back() = m_loaders[ti].fallback();
97  m_fallbackHandles[ti] = m_resources.size() - 1;
98  }
99  return m_fallbackHandles[ti];
100  }
101 
102  auto handle = m_resources.size() - 1;
103  m_resourcePaths.insert(std::make_pair(path, handle));
104 
105  return handle;
106  }
