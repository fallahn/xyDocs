74  T& get(const std::string& path = "default")
75  {
76  //if we have a valid path check current resources and return if found
77  if (!path.empty())
78  {
79  auto r = m_resources.find(path);
80  if (r != m_resources.end())
81  {
82  return *r->second;
83  }
84  }
85  //else attempt to load from file
86  std::unique_ptr<T> r = std::make_unique<T>();
87  if (path.empty() || !r->loadFromFile(path))
88  {
89  m_resources[path] = errorHandle();
90  }
91  else
92  {
93  m_resources[path] = std::move(r);
94  }
95 
96  return *m_resources[path];
97  }
98  protected:
106  virtual std::unique_ptr<T> errorHandle() = 0;
107  private:
108  std::unordered_map<std::string, std::unique_ptr<T>> m_resources;
109  };
