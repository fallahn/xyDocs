52  "#version 120\n"
53  "#define MAX_KERNEL_SIZE 64\n"
54 
55  "varying vec2 v_texCoord;\n"
56 
57  "uniform sampler2D u_positionMap;\n"
58  "uniform float u_sampleRadius = 1.5f;\n"
59  "uniform mat4 u_projectionMatrix;\n"
60  "uniform vec3 u_kernel[MAX_KERNEL_SIZE];\n"
61 
62  "void main()\n"
63  "{\n"
64  " vec3 pos = texture2D(u_positionMap, v_texCoord).xyz;\n"
65 
66  " float AO = 0.0;\n"
67 
68  " for (int i = 0; i < MAX_KERNEL_SIZE; ++i)\n"
69  " {\n"
70  " vec3 samplePos = pos + u_kernel[i];\n"
71  " vec4 offset = vec4(samplePos, 1.0);\n"
72  " offset = u_projectionMatrix * offset;\n"
73  " offset.xy /= offset.w;\n"
74  " offset.xy = offset.xy * 0.5 + vec2(0.5);\n"
75 
76  " float sampleDepth = texture2D(u_positionMap, offset.xy).b;\n"
77 
78  " if (abs(pos.z - sampleDepth) < u_sampleRadius)\n"
79  " {\n"
80  " AO += step(sampleDepth, samplePos.z);\n"
81  " }\n"
82  " }\n"
83 
84  " AO = AO / 64.0;\n"
85 
86  " gl_FragColor = vec4(vec3(clamp(AO * 2.0, 0.0, 1.0)), 1.0); //vec4(vec3(AO * AO), 1.0);//\n"

89  static const std::string SSAOFragment2 =
90  "#version 120\n"
91 
92  "varying vec2 v_texCoord;\n"
93 
94  "uniform sampler2D u_positionMap;\n"
95  "uniform sampler2D u_normalMap;\n"
96  "uniform sampler2D u_noiseMap;\n"
97  "uniform vec3 u_kernel[64];\n"
98  "uniform mat4 u_projectionMatrix;\n"
99 
100  "const int kernelSize = 64;\n"
101  "const float radius = 5.0;\n"
102 
103  "void main()\n"
104  "{\n"
105  " vec3 fragPos = texture2D(u_positionMap, v_texCoord).xyz;\n"
106  " vec3 normal = texture2D(u_normalMap, v_texCoord).rgb;\n"
107  " vec3 randomVec = texture2D(u_noiseMap, v_texCoord * vec2(960.0f / 4.0f, 540.0f / 4.0f)).xyz;\n"
108 
109  " vec3 tangent = normalize(randomVec - normal * dot(randomVec, normal))\n;"
110  " vec3 bitangent = cross(normal, tangent);\n"
111  " mat3 TBN = mat3(tangent, bitangent, normal);\n"
112 
113  " float occlusion = 0.0;\n"
114  " for (int i = 0; i < kernelSize; ++i)\n"
115  " {\n"
116  " vec3 sample = TBN * u_kernel[i];\n"
117  " sample = fragPos + sample * radius;\n"
118 
119  " vec4 offset = vec4(sample, 1.0);\n"
120  " offset = u_projectionMatrix * offset;\n"
121  " offset.xyz /= offset.w;\n"
122  " offset.xyz = offset.xyz * 0.5 + 0.5;\n"
123 
124  " float sampleDepth = -texture2D(u_positionMap, offset.xy).w;\n"
125 
126  " float rangeCheck = smoothstep(0.0, 1.0, radius / abs(fragPos.z - sampleDepth));\n"
127  " occlusion += (sampleDepth >= sample.z ? 1.0 : 0.0) * rangeCheck;\n"
128  " }"
129  " occlusion = 1.0 - (occlusion / kernelSize);\n"
130 
131  " gl_FragColor = vec4(vec3(occlusion), 1.0);\n"
132  "}";
