49  static inline std::vector<float> toFloatArray(const std::string& str)
50  {
51  std::vector<float> values;
52  std::size_t start = 0u;
53  auto next = str.find_first_of(',');
54  while (next != std::string::npos && start < str.length())
55  {
56  try
57  {

64  start = ++next;

140  static inline std::vector<uint32_t> getCodepoints(const std::string& str)
141  {
142  std::vector<uint32_t> codePoints;
143 
144  for (uint32_t i = 0; i < str.size(); ++i)
145  {
146  if ((str[i] >> 7) == 0)
147  {
148  codePoints.push_back(uint32_t(str[i]));
149  }
150  else if (((uint8_t)str[i] >> 5) == 6)
151  {
152  // add "x"s to the unicode bits //
153  uint8_t character = str[i + 1];
154  // take away the "10" //
155  uint8_t bitmask = 127; // 0x01111111
156  uint32_t codePoint = uint32_t(character & bitmask);
157 
158  // add "y"s to the unicode bits //
159  character = str[i];
160  // take away the "110" //
161  bitmask = 63; // 0x00111111
162  uint32_t yValues = uint32_t(character & bitmask);
163  codePoint = codePoint | (yValues << 6);
164 
165  codePoints.push_back(codePoint);
166  ++i;
167  }
168  else if (((uint8_t)str[i] >> 4) == 14)
169  {
170  // add "x"s to the unicode bits //
171  uint8_t character = str[i + 2];
172  // take away the "10" //
173  uint8_t bitmask = 127; // 0x01111111
174  uint32_t codePoint = uint32_t(character & bitmask);
175 
176  // add "y"s to the unicode bits //
177  character = str[i + 1];
178  // take away the "10" //
179  uint32_t yValues = uint32_t(character & bitmask);
180  codePoint = codePoint | (yValues << 6);
181 
182  // add "z"s to the unicode bits //
183  character = str[i];
184  // take away the "1110" //
185  bitmask = 31; // 0x00011111
186  uint32_t zValues = uint32_t(character & bitmask);
187  codePoint = codePoint | (zValues << 12);
188 
189  codePoints.push_back(codePoint);
190  i += 2;
191  }
192  else if (((uint8_t)str[i] >> 3) == 30)
193  {
194  // add "x"s to the unicode bits //
195  uint8_t character = str[i + 3];
196  // take away the "10" //
197  uint8_t bitmask = 127; // 0x01111111
198  uint32_t codePoint = uint32_t(character & bitmask);
199
